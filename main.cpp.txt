#include <iostream>
#include <fstream>
using namespace std;

class Student {
public:
    int id;
    string name;
    int age;
    float marks;
};

void addStudent() {
    Student s;
    ofstream file("students.txt", ios::app);

    cout<<"Enter ID: ";
    cin>>s.id;

    cout<<"Enter Name: ";
    cin>>s.name;

    cout<<"Enter Age: ";
    cin>>s.age;

    cout<<"Enter Marks: ";
    cin>>s.marks;

    file<<s.id<<" "<<s.name<<" "<<s.age<<" "<<s.marks<<endl;

    file.close();

    cout<<"Student added successfully\n";
}

void displayStudents() {
    Student s;
    ifstream file("students.txt");

    cout<<"\nStudent Records\n";

    while(file>>s.id>>s.name>>s.age>>s.marks) {
        cout<<s.id<<" "<<s.name<<" "<<s.age<<" "<<s.marks<<endl;
    }

    file.close();
}

int main() {

    int choice;

    do {
        cout<<"\n--- Student Management System ---\n";
        cout<<"1. Add Student\n";
        cout<<"2. Display Students\n";
        cout<<"3. Exit\n";
        cout<<"Enter choice: ";

        cin>>choice;

        switch(choice) {
            case 1:
                addStudent();
                break;

            case 2:
                displayStudents();
                break;

            case 3:
                cout<<"Exit\n";
                break;

            default:
                cout<<"Invalid choice\n";
        }

    } while(choice!=3);

    return 0;
}
